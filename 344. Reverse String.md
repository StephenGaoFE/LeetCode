var reverseString2 = function(s) {
	var arr = s.split('');
	for (var i=0, j= arr.length-1; i<j; ++i, --j){
		var tmp = arr[i];
		arr[i] = arr[j];
		arr[j] = tmp;
	}
	return arr.join('');
}
console.log(reverseString2('Bing'));
